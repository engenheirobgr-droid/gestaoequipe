        const formatTime = (ms) => {
             if (typeof ms !== 'number' || ms < 0) return '00:00:00';
             const totalSeconds = Math.floor(ms / 1000);
             const hours = Math.floor(totalSeconds / 3600).toString().padStart(2, '0');
             const minutes = Math.floor((totalSeconds % 3600) / 60).toString().padStart(2, '0');
             const seconds = (totalSeconds % 60).toString().padStart(2, '0');
             return `${hours}:${minutes}:${seconds}`;
        };

        const handleFileImport = (event, processDataCallback) => {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = (e) => {
                const data = e.target.result;
                let workbook;
                try {
                    workbook = XLSX.read(data, { type: 'binary', cellDates: true, codepage: 65001 });
                } catch (error) {
                    console.error("ERRO: Falha ao ler o arquivo.", error);
                    alert("ERRO: Não foi possível ler o arquivo. Verifique se o formato é válido.");
                    return;
                }

                const sheetName = workbook.SheetNames[0];
                const jsonData = XLSX.utils.sheet_to_json(workbook.Sheets[sheetName]);
                processDataCallback(jsonData);
            };
            reader.readAsBinaryString(file);
            event.target.value = '';
        }

         const handleExport = () => {

             const dataToExport = tasks.map(task => {
                 const exportRow = { ...task.originalData };

                 if (task.createdAt && task.finalizadoAt) {
                     exportRow['Lead Time (dias)'] = ((toValidDate(task.finalizadoAt) - toValidDate(task.createdAt)) / msPerDay).toFixed(1);
                 } else {
                     exportRow['Lead Time (dias)'] = 'N/A';
                 }

                 if (task.startedAt && task.studyCompletedAt) {
                    const cycleTimeMs = toValidDate(task.studyCompletedAt) - toValidDate(task.startedAt) - (task.waitingTime || 0);
                    exportRow['Cycle Time (dias)'] = (cycleTimeMs / msPerDay).toFixed(1);
                 } else {
                      exportRow['Cycle Time (dias)'] = 'N/A';
                 }

                 exportRow['Tempo Gasto (HH:MM:SS)'] = formatTime(task.timeSpent);
                 exportRow['Tempo de Espera (HH:MM:SS)'] = formatTime(task.waitingTime || 0);
                 exportRow['Contagem de Retrabalho'] = task.reworkCount || 0;
                 exportRow['Tempo Previsto (h)'] = task.expectedTime;

                 if (task.studyCompletedAt) exportRow['Data de Conclusão Estudo'] = toValidDate(task.studyCompletedAt);

         const sortTasks = (tasksToSort, sortBy, sortDirection) => {
             const priorityMap = { 'Alta': 3, 'Média': 2, 'Baixa': 1 };

             return [...tasksToSort].sort((a, b) => {
                 let primaryA, primaryB;

                 switch (sortBy) {
                     case 'score': primaryA = a.scoreGUT || a.scoreRICE || 0; primaryB = b.scoreGUT || b.scoreRICE || 0; break;
                     case 'createdAt': primaryA = toValidDate(a.createdAt); primaryB = toValidDate(b.createdAt); break;
                     case 'id': primaryA = parseInt(a.originalId, 10) || 0; primaryB = parseInt(b.originalId, 10) || 0; break;
                     case 'reclamacoes': primaryA = a.reclamacoes30d || 0; primaryB = b.reclamacoes30d || 0; break;
                     case 'criticidade': primaryA = priorityMap[a.originalData?.Prioridade] || 0; primaryB = priorityMap[b.originalData?.Prioridade] || 0; break;
                     default: primaryA = a.scoreGUT || a.scoreRICE || 0; primaryB = b.scoreGUT || b.scoreRICE || 0;
                 }

                 const direction = sortDirection === 'asc' ? 1 : -1;
                 if (primaryA < primaryB) return -1 * direction;
                 if (primaryA > primaryB) return 1 * direction;

                 const scoreA = a.scoreGUT || a.scoreRICE || 0;
                 const scoreB = b.scoreGUT || b.scoreRICE || 0;
                 if (scoreA !== scoreB) return scoreB - scoreA;

                 const reclamacoesA = a.reclamacoes30d || 0;
                 const reclamacoesB = b.reclamacoes30d || 0;
                 if (reclamacoesA !== reclamacoesB) return reclamacoesB - reclamacoesA;

                 const dateA = toValidDate(a.createdAt);
                 const dateB = toValidDate(b.createdAt);
                 if (dateA && dateB) return dateA - dateB;
                 return 0;
             });

         const generateHistoricalCfdData = (currentTasks) => {
             if (currentTasks.length === 0) return [];
             const validCreationDates = currentTasks.map(t => toValidDate(t.createdAt)).filter(d => d);
             if (validCreationDates.length === 0) return [];
             
             const earliestDate = new Date(Math.min(...validCreationDates));
             const today = new Date();
             let historyData = [];

             for (let d = new Date(earliestDate); d <= today; d.setDate(d.getDate() + 1)) {
                 const currentDate = new Date(d); currentDate.setHours(23, 59, 59, 999);
                 const dailyCounts = Object.keys(STATUS_OPTIONS).reduce((acc, status) => ({ ...acc, [status]: 0 }), {});
                 
                 currentTasks.forEach(task => {
                     const taskCreatedAt = toValidDate(task.createdAt);
                     if (!taskCreatedAt || taskCreatedAt > currentDate) return;
                     
                     const taskFinalizadoAt = toValidDate(task.finalizadoAt);
                     if(taskFinalizadoAt && taskFinalizadoAt <= currentDate) {
                         dailyCounts.finalizado++;
                     } else {
                         dailyCounts[task.status]++;
                     }
                 });
                 historyData.push({ date: currentDate.toISOString().split('T')[0], ...dailyCounts });
             }
             return historyData;

        const handleGlobalPlay = (taskId, subtaskId) => {

            const activeParticipantsInTask = [taskToPlay.responsible, ...(taskToPlay.collaborators?.filter(c => c.isActive).map(c => c.name) || [])].filter(Boolean);

            for (const name of activeParticipantsInTask) {
                const conflictingTask = tasks.find(task => {
                    if (task.id === taskId) return false; // Não comparar com a própria tarefa

                    const isTiming = task.subtasks && task.subtasks.some(st => st.isTiming);
                    if (!isTiming) return false;

                    const allActiveParticipants = [task.responsible, ...(task.collaborators?.filter(c => c.isActive).map(c => c.name) || [])];
                    return allActiveParticipants.includes(name);
                });

                if (conflictingTask) {
                    alert(`Ação bloqueada: O participante "${name}" já está em uma sessão de trabalho ativa na demanda "${conflictingTask.title}".`);
                    return;

            }
            // FIM DA ALTERAÇÃO

            handlePlay(taskId, subtaskId);

        const handlePlay = async (taskId, subtaskId) => {
    const task = tasks.find(t => t.id === taskId);
    const subtaskIndex = task?.subtasks.findIndex(st => st.id === subtaskId);
    if (!task || subtaskIndex === -1 || task.subtasks[subtaskIndex].isTiming) return;

    const updatedSubtasks = [...task.subtasks];
    updatedSubtasks[subtaskIndex].isTiming = true;
    updatedSubtasks[subtaskIndex].lastStartTime = new Date().toISOString();

    let dataToUpdate = {
        subtasks: updatedSubtasks,
        playCount: (task.playCount || 0) + 1,
        lastActiveSubtaskId: subtaskId,
        collaborators: task.collaborators // <-- CORREÇÃO: Adiciona o estado atual dos colaboradores à atualização
    };
    
    if (task.status !== 'estudo_andamento' && task.status !== 'acao_andamento') {
        dataToUpdate.status = 'estudo_andamento';
        dataToUpdate.originalData = { ...task.originalData, 'Status': STATUS_TO_SHEET_MAP['estudo_andamento'] };
        
        if (isEditingModal && elements.modal.id.value === taskId) {
            elements.modal.status.value = 'estudo_andamento';
        }
    }

    if (!task.startedAt) {
        dataToUpdate.startedAt = new Date(updatedSubtasks[subtaskIndex].lastStartTime);
    }

    const taskRef = doc(db, 'dados_da_equipe', ID_DA_EQUIPE, 'tasks', taskId);
    await updateDoc(taskRef, dataToUpdate); // <-- AGORA ENVIA O PACOTE COMPLETO

    if(isEditingModal && elements.modal.id.value === taskId) {
        const updatedTask = { ...task, ...dataToUpdate };
        renderSubtasks(updatedTask);
    }
};

        const handlePause = async (taskId, subtaskId, options = {}) => {
    const task = tasks.find(t => t.id === taskId);
    const subtaskIndex = task?.subtasks.findIndex(st => st.id === subtaskId);
    if (!task || subtaskIndex === -1 || !task.subtasks[subtaskIndex].isTiming) return;
    
    const elapsed = new Date() - toValidDate(task.subtasks[subtaskIndex].lastStartTime);
    
    const activeCollaborators = task.collaborators?.filter(c => c.isActive) || [];
    const activeParticipantCount = 1 + activeCollaborators.length;
    const timeToAdd = elapsed * activeParticipantCount;

    const updatedSubtasks = [...task.subtasks];
    updatedSubtasks[subtaskIndex].timeSpent += timeToAdd;
    updatedSubtasks[subtaskIndex].isTiming = false;
    updatedSubtasks[subtaskIndex].lastStartTime = null;

    let totalTimeSpent = updatedSubtasks.reduce((total, sub) => total + sub.timeSpent, 0);
    
    const dataToUpdate = {
        subtasks: updatedSubtasks,

                            obj[key] = obj[key].toDate();
                        } else if (typeof obj[key] === 'object' && obj[key] !== null) {
                            convertTimestamps(obj[key]);
                        }
                    }
                };
                convertTimestamps(docData);

                const legacyStatusMap = { 'em andamento': 'estudo_andamento', 'em_andamento': 'estudo_andamento', 'em validacao': 'estudo_validacao', 'em_validacao': 'estudo_validacao','validacao': 'estudo_validacao', 'andamento': 'estudo_andamento','concluido': 'estudo_concluido', 'concluído': 'estudo_concluido', 'concluida': 'estudo_concluido', 'concluída': 'estudo_concluido' };
                const oldStatus = docData.status;
                if (legacyStatusMap[oldStatus]) {
                    docData.status = legacyStatusMap[oldStatus];
                    if ((oldStatus.includes('andamento') || oldStatus.includes('validacao')) && !docData.startedAt) {
                        docData.startedAt = docData.createdAt;
                    }
                    if (oldStatus.includes('concluid')) {
                        if (!docData.startedAt) docData.startedAt = docData.createdAt;
                        if (!docData.studyCompletedAt) docData.studyCompletedAt = docData.completedAt;
                    }
                }
                return docData;
            };

            let tasksQuery;
            if (currentUserRole === 'manager') {
                tasksQuery = query(tasksCollectionRef);
            } else {
                tasksQuery = query(tasksCollectionRef, where("isPrivate", "==", false));
            }

            unsubscribeTasks = onSnapshot(tasksQuery, (snapshot) => {
                let needsFullRerender = false;
                let headersNeedUpdate = false;

                snapshot.docChanges().forEach((change) => {
                    needsFullRerender = true;
                    const changedData = processTaskData(change.doc.data());
                    const taskId = change.doc.id;

                    if (change.type === "added") {
                        if (!tasks.some(t => t.id === taskId)) {
                           tasks.push({ id: taskId, ...changedData });
                           headersNeedUpdate = true;
                        }
                    }
                    if (change.type === "modified") {
                        const index = tasks.findIndex(t => t.id === taskId);
                        if (index > -1) {
                            tasks[index] = { id: taskId, ...changedData };
                        }
                    }
                    if (change.type === "removed") {
                        tasks = tasks.filter(t => t.id !== taskId);
                    }
                });

                    addedFields.forEach(h => newHeaders.add(h));

                        if (task.originalData) {
                            Object.keys(task.originalData).forEach(key => newHeaders.add(key));
                        }
                    });
                    allHeaders = Array.from(newHeaders);
                    populateGlobalFilters();
                }

                if (needsFullRerender) {
                    renderActiveTabView();

        const positionTooltip = (container) => {
            const tooltip = container.querySelector('.tooltip');
            if (!tooltip) return;

            const iconRect = container.getBoundingClientRect();
            tooltip.style.visibility = 'visible';
            const tooltipHeight = tooltip.offsetHeight;
            const tooltipWidth = tooltip.offsetWidth;
            tooltip.style.visibility = ''; 
            const spaceAbove = iconRect.top;
            const spaceBelow = window.innerHeight - iconRect.bottom;
            let top, left;

            if (spaceAbove < tooltipHeight + 8 && spaceBelow > spaceAbove) {
                top = iconRect.bottom + 8;
            } else {
                top = iconRect.top - tooltipHeight - 8;
            }
            left = iconRect.left + (iconRect.width / 2) - (tooltipWidth / 2);
            if (left < 8) left = 8;
            if (left + tooltipWidth > window.innerWidth - 8) {
                left = window.innerWidth - tooltipWidth - 8;
            }
            tooltip.style.top = `${top}px`;
            tooltip.style.left = `${left}px`;
        };

                else { alimentadoresSort.by = key; alimentadoresSort.direction = 'asc'; }
                renderAlimentadoresTable();
            });

            document.getElementById('add-user-form').addEventListener('submit', async (e) => {
                e.preventDefault();
                const matricula = document.getElementById('new-user-matricula').value.trim();
                const nome = document.getElementById('new-user-name').value.trim();
                const role = document.getElementById('new-user-role').value;
            
                if (!matricula || !nome) {
                    alert("Matrícula e Nome são obrigatórios.");
                    return;
                }
            
                const userRef = doc(db, 'allowed_users', matricula);
                await setDoc(userRef, { nome, role });
                
                alert(`Usuário ${nome} (${matricula}) adicionado à lista de permissões.`);
                e.target.reset();
            });
            
            document.getElementById('allowed-users-table').addEventListener('click', async (e) => {
                if (e.target.classList.contains('remove-user-btn')) {
                    const matricula = e.target.dataset.matricula;
                    if (confirm(`Tem certeza que deseja remover a permissão da matrícula ${matricula}?`)) {
                        const userRef = doc(db, 'allowed_users', matricula);
                        await deleteDoc(userRef);
                        alert(`Permissão para ${matricula} removida.`);
                    }
                }
            });

         };

        renderGUTGuide();
        initAuthentication();
    });
